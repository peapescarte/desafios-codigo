#include <iostream>
#include <string>
#include <cctype>
using namespace std;

int main()
{
	setlocale(LC_ALL, "portuguese");
    string palavra, inv;
    cout << "Digite a palavra: ";
    cin >> palavra;
    
    for (int i = 0; i < palavra.size(); i++) {
        palavra[i] = tolower(palavra[i]);
    }

    for (int i = palavra.size() - 1; i >= 0; i--) {
        inv += palavra[i];
    }
 
    if (palavra == inv) {
        cout << "Esta palavra é palíndromo";
    } else {
        cout << "Esta palavra não é palíndromo";
    }

    return 0;
}
